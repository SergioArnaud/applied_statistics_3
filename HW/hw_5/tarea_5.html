def get_roc_auc(model, X_test, y_test):
    '''
    Funcin que caclula y regresa el area bajo la curva ROC
    '''
    fpr, tpr, threshold = roc_curve(y_test, model.predict(X_test))
    roc_auc = auc(fpr, tpr)
    return roc_auc


def evaluate_model(model, data, target, n=10, test_size=.2):
    '''
    Funcin que recibe un modelo y los datos de entrenamiento. Para n iteraciones se Obtiene una muestra 
    del (1-test_size)*100% para la cual ajusta dicho modelo y posteriormente lo evala en los datos
    restantes. Se regresan el rea promedio bajo la curva ROC, el score de cross-validation promedio,
    la matriz de confusin promedio y el error aparente.
    '''
    scores, matrices, roc_auc  = [],[],[]
    for i in range(n):
        
        # Partitioning train and test data
        X_train, X_test, y_train, y_test = train_test_split(data, target, test_size=test_size)
        
        # Fitting model
        model.fit(X_train, y_train)
        
        # Obtaining Cross-validation score
        scores.append(model.score(X_test, y_test['loan_status']) )
        
        # Obtaining confusion matrix
        matrices.append(confusion_matrix(y_test['loan_status'], model.predict(X_test)))
        
        # Obtaining area under the ROC curve
        roc_auc.append(get_roc_auc(model, X_test, y_test))
        
        
    # Calculating average confusion matrix
    matrix =  np.mean(matrices,axis=0)
    
    # Obtaining APER
    APER = 100*(matrix[0,1]+matrix[1,0])/sum(sum(matrix))
    
    d = {'ROC-AUC': [np.mean(roc_auc)], 
         'CV-score': [np.mean(scores)], 
         'Confusion-Matrix':[matrix], 
         'Covariance-Matrix':[np.cov(matrix).round(1)],
         'APER':[APER]}
    
    return pd.DataFrame(data=d)

def plot_confusion_matrix(cm, classes,normalize=False,title='Confusion matrix',cmap=plt.cm.Blues):
    """
    This function prints and plots the confusion matrix.
    Normalization can be applied by setting `normalize=True`.
    """
    if normalize:
        cm = cm.astype('float') / cm.sum(axis=1)[:, np.newaxis]

    plt.imshow(cm, interpolation='nearest', cmap=cmap)
    plt.title(title)
    plt.colorbar()
    tick_marks = np.arange(len(classes))
    plt.xticks(tick_marks, classes, rotation=45)
    plt.yticks(tick_marks, classes)

    fmt = '.3f'
    thresh = cm.max() / 2.
    for i, j in itertools.product(range(cm.shape[0]), range(cm.shape[1])):
        plt.text(j, i, format(cm[i, j], fmt),
                 horizontalalignment="center",
                 color="white" if cm[i, j] > thresh else "black")

    plt.ylabel('True label')
    plt.xlabel('Predicted label')
    plt.tight_layout()

La respuesta es sencilla, tenemos clases sumamente desbalanceadas (caso el 85% de los prstamos fueron pagados en tiempo y forma) de forma que al entrenar los modelos se tiene una tendencia mucho mayor a clasificar el prestamo como uno pagato en tiempo y forma.

Cmo solucionarlo? El mtodo ms comun -y el que usaremos en este caso- es el de oversampling que consiste balancear las clases artificialmente (haciendo uso de un mtodo llamado SMOTE) y posteriormente entrenar el modelo.

def evaluate_model_2(model, data, target, n=10, test_size=.2):
    '''
    Funcin que recibe un modelo y los datos de entrenamiento. Para n iteraciones se Obtiene una muestra 
    del (1-test_size)*100% para la cual ajusta dicho modelo y posteriormente lo evala en los datos
    restantes. Se regresan el rea promedio bajo la curva ROC, el score de cross-validation promedio,
    la matriz de confusin promedio y el error aparente.
    '''
    scores, matrices, roc_auc  = [],[],[]
    for i in range(n):
        
        # Partitioning train and test data
        X_train, X_test, y_train, y_test = train_test_split(data, target, test_size=test_size)

        sm = SMOTE()
        X_train, y_train = sm.fit_sample(X_train, y_train['loan_status'].ravel())
        
        # Fitting model
        model.fit(X_train, y_train)
        
        # Obtaining Cross-validation score
        scores.append(model.score(X_test, y_test['loan_status']) )
        
        # Obtaining confusion matrix
        matrices.append(confusion_matrix(y_test['loan_status'], model.predict(X_test)))
        
        # Obtaining area under the ROC curve
        roc_auc.append(get_roc_auc(model, X_test, y_test))
        
        
    # Calculating average confusion matrix
    matrix =  np.mean(matrices,axis=0)
    
    # Obtaining APER
    APER = 100*(matrix[0,1]+matrix[1,0])/sum(sum(matrix))
    
    d = {'ROC-AUC': [np.mean(roc_auc)], 
         'CV-score': [np.mean(scores)], 
         'Confusion-Matrix':[matrix], 
         'Covariance-Matrix':[np.cov(matrix).round(1)],
         'APER':[APER]}
    
    return pd.DataFrame(data=d)
